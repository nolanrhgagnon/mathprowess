    \begin{enumerate}
    \item Evaluate the integral \(\displaystyle \int (3x^4 - 5x^2 + 2) \, dx\).
    \item Compute \(\displaystyle \int \frac{1}{x \ln x} \, dx\), for \(x > 1\).
    \item Find the area under the curve \(y = \sin(2x)\) from \(x=0\) to \(x=\pi\).
    \item Calculate the integral \(\displaystyle \int e^{3x} \cos(2x) \, dx\).
    \item Evaluate \(\displaystyle \int \frac{2x}{x^2 + 1} \, dx\).
    \item Determine \(\displaystyle \int \tan^3(x) \sec^2(x) \, dx\).
    \item Find \(\displaystyle \int \frac{1}{\sqrt{1 - x^2}} \, dx\).
    \item Compute the definite integral \(\displaystyle \int_0^1 x e^{x^2} \, dx\).
    \item Evaluate \(\displaystyle \int \frac{\ln(x)}{x} \, dx\).
    \item Calculate \(\displaystyle \int \frac{dx}{x^2 \sqrt{x^2 - 1}}\), for \(x > 1\).
\end{enumerate}
